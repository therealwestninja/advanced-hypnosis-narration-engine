  let tagSet = new Set(profileTags.map(t => String(t).toLowerCase()));
  let filtered = bank.filter(m => {
    if(!m || exclude.has(m.id)) return false;
    if(!m.tags) return true; // untagged motifs are universal
    let mTags = String(m.tags).split('|').map(t => t.trim().toLowerCase());
    return mTags.some(t => tagSet.has(t));
  });
  // Fallback: if the tag filter is too thin (cold-start, empty profileTags),
  // loosen to ANY non-excluded motif so callers always get something.
  if(filtered.length < n) {
    let seen = new Set(filtered.map(m => m.id));
    for(let m of bank) {
      if(m && !exclude.has(m.id) && !seen.has(m.id)) { filtered.push(m); seen.add(m.id); }
    }
  }
  // Fisher-Yates shuffle, then take first n. Return shallow copies so a caller
  // cannot mutate the shared corpus.
  let arr = filtered.slice();
  for(let i = arr.length - 1; i > 0; i--) {
    let j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr.slice(0, n).map(m => ({ ...m }));
