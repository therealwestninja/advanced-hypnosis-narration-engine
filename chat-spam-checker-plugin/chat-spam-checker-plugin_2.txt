<!-- ───────────────────────── demo wiring ─────────────────────────
  This page IS the plugin page, so root.$output(...) is the entry point.
  We use it directly to demo the guard. Consumers would import this page
  and alias it as e.g. spamChecker, then call root.spamChecker(...).
-->
<script>
(() => {
  const demoChannels = ["general", "off-topic", "help", "random"];

  const grid    = document.getElementById("channelsGrid");
  const chanSel = document.getElementById("chanSel");
  const userSel = document.getElementById("userSel");
  const msgIn   = document.getElementById("msgInput");
  const sendBtn = document.getElementById("sendBtn");
  const clearBtn= document.getElementById("clearBtn");
  const logEl   = document.getElementById("logEl");

  const channelEls = {};
  for (const name of demoChannels) {
    const div = document.createElement("div");
    div.className = "channel";
    div.innerHTML = '<div class="channel-head">#' + name + '</div><div class="channel-feed" data-feed="' + name + '"></div>';
    grid.appendChild(div);
    channelEls[name] = div.querySelector(".channel-feed");
    const opt = document.createElement("option");
    opt.value = name; opt.textContent = "#" + name;
    chanSel.appendChild(opt);
  }

  const log = (msg, cls) => {
    const line = document.createElement("div");
    if (cls) line.className = cls;
    line.textContent = "[" + new Date().toLocaleTimeString() + "] " + msg;
    logEl.appendChild(line);
    logEl.scrollTop = logEl.scrollHeight;
  };

  // The guard is created by calling THIS PAGE's own $output function.
  // Consumers of the plugin would write `root.spamChecker(...)` — but
  // here on the plugin page itself we call `root.$output(...)` directly.
  let guard;
  try {
    guard = root.$output({
      channelThreshold: 3,
      windowMs: 30000,
      whitelistUserIds: ["mod"],
      banFn: (userId, info) => {
        log("BAN " + userId + " (matched in " + info.spam.channels.length + " channels)", "ev-ban");
      },
      deleteFn: (messageId, channel) => {
        const feed = channelEls[channel];
        if (!feed) return;
        const el = feed.querySelector('[data-mid="' + messageId + '"]');
        if (el) {
          el.classList.add("deleted");
          el.classList.remove("banned");
          el.textContent = "[message removed by spam-guard]";
          log("CLEAR " + messageId + " from #" + channel, "ev-del");
        }
      },
      onSpamDetected: (report) => {
        log("SPAM PATTERN from " + (report.username || report.userId) + " across " + report.channels.length + " channels", "ev-ok");
      },
    });
  } catch (e) {
    log("FAILED to create guard: " + e.message, "ev-ban");
    console.error(e);
    return;
  }

  let msgCounter = 0;
  function postMessage() {
    const userId = userSel.value;
    const username = userSel.options[userSel.selectedIndex].textContent.split(" ")[0];
    const channel = chanSel.value;
    const text = msgIn.value.trim();
    if (!text) return;
    const messageId = "m" + (++msgCounter);

    const escapeHtml = s => s.replace(/[<>&]/g, c => ({"<":"&lt;",">":"&gt;","&":"&amp;"}[c]));
    const feed = channelEls[channel];
    const row = document.createElement("div");
    row.className = "msg";
    row.dataset.mid = messageId;
    row.innerHTML = '<span class="who">' + escapeHtml(username) + ':</span> ' + escapeHtml(text);
    feed.appendChild(row);
    feed.scrollTop = feed.scrollHeight;

    Promise.resolve(guard.report({ channel, userId, username, messageId, text }))
      .then(result => {
        if (result.banned) {
          document.querySelectorAll(".msg").forEach(el => {
            const who = el.querySelector(".who");
            if (who && who.textContent.replace(":","").trim() === username) {
              el.classList.add("banned");
            }
          });
        } else if (result.whitelisted) {
          log("OK " + username + " is whitelisted — skipping check", "ev-ok");
        }
      });

    msgIn.value = "";
    msgIn.focus();
  }

  sendBtn.addEventListener("click", postMessage);
  msgIn.addEventListener("keydown", e => { if (e.key === "Enter") postMessage(); });
  clearBtn.addEventListener("click", () => {
    Object.values(channelEls).forEach(el => { el.innerHTML = ""; });
    logEl.innerHTML = "";
    guard.flush();
    guard.bannedUsers().forEach(id => guard.unban(id));
    log("demo state cleared");
  });

  log("demo ready — try posting the same text from Eve into 3+ channels within 30s");
})();
</script>
