$meta
  title = Chat Spam Checker Plugin
  description = A drop-in cross-channel spam guard for the Perchance comments-plugin family (comments-plugin, tabbed-comments-plugin-v1, or any custom chat). Detects the canonical copy-paste spam pattern — the same or near-identical message posted to several different channels by the same user inside a short time window — then automatically bans the user and clears all of their flagged messages. Designed for AI chat apps, character-chat generators, roleplay pages, RP communities, and any Perchance page running multi-channel comments that is getting flooded by spammers posting Discord invites, crypto promos, NSFW URLs, scam links, or affiliate codes across every channel. Detection uses a normalised text fingerprint (lowercase, whitespace collapsed, punctuation and emoji stripped, repeated characters dampened, Unicode diacritics removed) plus a token-set Jaccard similarity check, so spammers tweaking their text by a word or two still get caught. Configurable channel threshold, sliding time window, similarity strictness, minimum message length, per-user history cap, optional ban duration for temporary mutes, moderator whitelist for accounts that legitimately cross-post announcements, and an autoBan toggle if you would rather surface spam reports for human review instead of auto-action. The plugin never bans or deletes anything by itself — you supply banFn and deleteFn callbacks that talk to your moderation backend or your generator commentOptions bannedUsers list, so the spam guard works with whatever permission model and admin password setup your generator already has. State is in-memory only by default so per-user histories reset on page reload, but the bans you persist via banFn stick. The page exposes one function which returns a guard object with report, ban, unban, isBanned, bannedUsers, history, flush, options, and stats methods. Try the live demo on this page before importing into your own generator.

// ══════════════════════════════════════════════════════════════════════
// chat-spam-checker-plugin · top zone
// ══════════════════════════════════════════════════════════════════════
//
// QUICK START — paste this into the top zone of your own generator:
//
//   spamChecker = {import:chat-spam-checker-plugin}
//
// THEN, in your HTML panel, create a guard and start feeding it messages:
//
//   let guard = root.spamChecker({
//     channelThreshold: 3,                         // ban after the same msg appears in 3+ channels
//     windowMs: 60000,                             // sliding window (60s)
//     similarityThreshold: 0.92,                   // 0..1 — fuzzy match strictness
//     minMessageLength: 5,                         // ignore "hi", "lol", etc.
//     whitelistUserIds: [moderatorUserId],         // these users are never flagged
//     banFn:    (userId)              => addToBanList(userId),
//     deleteFn: (messageId, channel)  => removeMessageOnBackend(messageId, channel),
//   });
//
//   // Then, every time a new message lands in any channel, feed it in:
//   guard.report({
//     channel:   "general",
//     userId:    "abc123",
//     username:  "Alice",
//     messageId: "msg_42",
//     text:      "Check out my crypto site!!!",
//   });
//
// Full options table, returned guard API, comments-plugin wiring recipe,
// and a live interactive demo are all rendered on this page (HTML panel).
//
// ══════════════════════════════════════════════════════════════════════
// IMPLEMENTATION NOTE
// ══════════════════════════════════════════════════════════════════════
// The full implementation lives INSIDE this $output function. The reason:
// when other generators import this page via {import:chat-spam-checker-plugin},
// only the top zone DSL is loaded — the HTML panel script does NOT run on
// the host. So the implementation has to live in the top-zone function
// closure. Every call to root.spamChecker(opts) creates a fresh guard with
// its own private state (history Map, banned Map, stats counter), so
// multiple independent guards can coexist on the same page.
//
// Perchance convention: the page's callable must be named $output — that's
// the function that gets invoked when consumers call their imported alias.
// So the host's `root.spamChecker(opts)` actually invokes our `$output(opts)`.
// ══════════════════════════════════════════════════════════════════════
$output(opts) =>
  if(!opts) opts = {};

  // ── DEFAULTS ────────────────────────────────────────────────────────
  const DEFAULTS = {
    channelThreshold: 3,
    windowMs: 60000,
    similarityThreshold: 0.92,
    minMessageLength: 5,
    maxHistoryPerUser: 50,
    autoBan: true,
    autoDelete: true,
    banDurationMs: 0,
    whitelistUserIds: [],
    banFn: null,
    deleteFn: null,
    onSpamDetected: null,
    onBan: null,
  };
  const merged = Object.assign({}, DEFAULTS, opts);

  // ── HELPERS ─────────────────────────────────────────────────────────
  // Normalisation: lowercases, strips emoji + punctuation + diacritics,
  // dampens repeated characters, collapses whitespace. Two messages with
  // the same fingerprint are considered identical for spam-detection.
  function fingerprint(text) {
    if (typeof text !== "string") return "";
    return text
      .toLowerCase()
      .normalize("NFKD")
      .replace(/[\u0300-\u036f]/g, "")
      .replace(/[\p{Emoji_Presentation}\p{Extended_Pictographic}]/gu, "")
      .replace(/[^\p{L}\p{N}\s]/gu, " ")
      .replace(/(.)\1{2,}/g, "$1$1")
      .replace(/\s+/g, " ")
      .trim();
  }

  // Jaccard similarity on word-token sets — handles "near-identical"
  // messages where one or two words differ. Range: 0 (disjoint) → 1 (same).
  function jaccard(a, b) {
    if (a === b) return 1;
    if (!a || !b) return 0;
    const A = new Set(a.split(" "));
    const B = new Set(b.split(" "));
    let inter = 0;
    for (const t of A) if (B.has(t)) inter++;
    const union = A.size + B.size - inter;
    return union === 0 ? 0 : inter / union;
  }

  function similarEnough(a, b, threshold) {
    if (a === b) return true;
    return jaccard(a, b) >= threshold;
  }

  // ── PER-GUARD PRIVATE STATE ─────────────────────────────────────────
  const history = new Map();   // userId -> [{channel, fp, text, messageId, timestamp}]
  const banned  = new Map();   // userId -> {reason, bannedAt, expiresAt|null}
  const stats   = { tracked: 0, banned: 0, totalReports: 0, totalBans: 0, totalDeletes: 0 };

  function pruneOld(userId, now) {
    const arr = history.get(userId);
    if (!arr) return;
    const cutoff = now - merged.windowMs;
    let i = 0;
    while (i < arr.length && arr[i].timestamp < cutoff) i++;
    if (i > 0) arr.splice(0, i);
    if (arr.length > merged.maxHistoryPerUser) {
      arr.splice(0, arr.length - merged.maxHistoryPerUser);
    }
  }

  function checkBanExpiry(userId, now) {
    const b = banned.get(userId);
    if (b && b.expiresAt && b.expiresAt <= now) {
      banned.delete(userId);
      return false;
    }
    return !!b;
  }

  // Returns the spam report object if this message triggers a ban,
  // or null otherwise. Keeps EARLIEST match per channel so deletion
  // reaches the original message, not the latest one.
  function classify(userId, fp, now) {
    const arr = history.get(userId) || [];
    const matchedByChannel = new Map();
    for (const m of arr) {
      if (similarEnough(m.fp, fp, merged.similarityThreshold)) {
        if (!matchedByChannel.has(m.channel)) matchedByChannel.set(m.channel, m);
      }
    }
    if (matchedByChannel.size >= merged.channelThreshold) {
      const matches = Array.from(matchedByChannel.values());
      return {
        fingerprint: fp,
        channels: Array.from(matchedByChannel.keys()),
        messages: matches,
        firstSeen: Math.min.apply(null, matches.map(m => m.timestamp)),
        lastSeen:  now,
      };
    }
    return null;
  }

  // ── PUBLIC: report a new message ───────────────────────────────────
  async function report(input) {
    stats.totalReports++;
    if (!input || !input.userId || !input.channel) {
      return { ok: false, reason: "missing userId or channel" };
    }
    const channel   = input.channel;
    const userId    = input.userId;
    const username  = input.username;
    const messageId = input.messageId;
    const text      = input.text;
    if (merged.whitelistUserIds.indexOf(userId) !== -1) {
      return { ok: true, whitelisted: true };
    }

    const now = input.timestamp || Date.now();
    if (checkBanExpiry(userId, now)) {
      return { ok: false, banned: true };
    }

    const fp = fingerprint(text);
    if (!fp || fp.length < merged.minMessageLength) {
      return { ok: true, ignored: "too-short" };
    }

    if (!history.has(userId)) {
      history.set(userId, []);
      stats.tracked++;
    }
    history.get(userId).push({ channel, fp, text, messageId, timestamp: now });
    pruneOld(userId, now);

    const spam = classify(userId, fp, now);
    if (!spam) return { ok: true };

    spam.userId = userId;
    spam.username = username;
    if (spam.channels.indexOf(channel) === -1) {
      spam.channels.push(channel);
      spam.messages.push({ channel, fp, text, messageId, timestamp: now });
    }

    if (typeof merged.onSpamDetected === "function") {
      const verdict = await merged.onSpamDetected(spam);
      if (verdict === false) return { ok: true, vetoed: true, spam };
    }

    if (merged.autoBan) {
      const expiresAt = merged.banDurationMs > 0 ? now + merged.banDurationMs : null;
      banned.set(userId, { reason: "cross-channel-spam", bannedAt: now, expiresAt });
      stats.banned++;
      stats.totalBans++;
      if (typeof merged.banFn === "function") {
        try { await merged.banFn(userId, { reason: "cross-channel-spam", spam }); }
        catch (e) { console.error("[spam-guard] banFn threw:", e); }
      }
      if (typeof merged.onBan === "function") {
        try { await merged.onBan(userId, { reason: "cross-channel-spam", spam }); } catch (e) {}
      }
    }
    if (merged.autoDelete && typeof merged.deleteFn === "function") {
      for (const m of spam.messages) {
        try {
          await merged.deleteFn(m.messageId, m.channel);
          stats.totalDeletes++;
        } catch (e) { console.error("[spam-guard] deleteFn threw:", e, m); }
      }
    }
    history.delete(userId);
    return { ok: false, banned: true, spam };
  }

  // ── PUBLIC: manual controls ────────────────────────────────────────
  function ban(userId, reason) {
    banned.set(userId, { reason: reason || "manual", bannedAt: Date.now(), expiresAt: null });
    stats.banned++; stats.totalBans++;
  }
  function unban(userId) {
    if (banned.delete(userId)) stats.banned = Math.max(0, stats.banned - 1);
  }
  function isBanned(userId) { return checkBanExpiry(userId, Date.now()); }
  function bannedUsers() {
    const now = Date.now();
    return Array.from(banned.keys()).filter(id => checkBanExpiry(id, now));
  }
  function getHistory(userId) { return (history.get(userId) || []).slice(); }
  function flush() { history.clear(); }
  function getStats() {
    return Object.assign({}, stats, { tracked: history.size, banned: bannedUsers().length });
  }

  return {
    report, ban, unban, isBanned, bannedUsers,
    history: getHistory, flush, options: merged, stats: getStats,
    _internal: { history, banned, fingerprint, jaccard },
  };
